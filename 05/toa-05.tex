\begin{align*}
Fib(0) =& 1\\
Fib(1) =& 1\\
Fib(n) =& Fib(n - 1) + Fib(n - 2)\\
\end{align*}

\begin{align*}
A(0, n) =& n + 1\\
A(m + 1, 0) =& A(m, 1)\\
A(m + 1, n + 1) =& A(m, A(m + 1, n))
\end{align*}
